        <input type="text" id="employeeName" placeholder="例如：1號 小明">
        <br>
        <button onclick="clockIn()">上班打卡</button>

            <h3>📋 今日打卡紀錄</h3>

    <script>
        // 載入時顯示過去的紀錄
        window.onload = function() {
            displayRecords();
        };

        function clockIn() {
            const nameInput = document.getElementById('employeeName');
            const name = nameInput.value.trim();

            if (name === "") {
                alert("請先輸入員工姓名或座號喔！");
                return;
            }

            const now = new Date();
            const timeString = now.toLocaleDateString('zh-TW') + " " + now.toLocaleTimeString('zh-TW');
            const record = `✅ ${name} - 於 ${timeString} 完成打卡`;

            // 取得舊紀錄，加入新紀錄後存回 LocalStorage
            let records = JSON.parse(localStorage.getItem('classPunchRecords')) || [];
            records.unshift(record); // 將新紀錄加到最前面
            localStorage.setItem('classPunchRecords', JSON.stringify(records));

            nameInput.value = ""; // 清空輸入框
            displayRecords();
            
            alert(`${name}，打卡成功！祝你有個美好的一天！`);
        }

        function displayRecords() {
            const recordList = document.getElementById('recordList');
            const records = JSON.parse(localStorage.getItem('classPunchRecords')) || [];
            
            recordList.innerHTML = "";
            if(records.length === 0) {
                recordList.innerHTML = "<p style='color:#999;'>目前還沒有人打卡喔。</p>";
                return;
            }

            records.forEach(record => {
                const div = document.createElement('div');
                div.className = 'record-item';
                div.innerText = record;
                recordList.appendChild(div);
            });
        }
    </script>
